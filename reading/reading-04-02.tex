\begin{problem}If $m/n=k/l$, where $m,n,k,l\in\mathbb{N}$
  does $(x^{1/n})^m=(x^{1/l})^k$?
\end{problem}
\begin{solution}\begin{align*}
  (x^{1/n})^m
    &= x^{m/n}\\
    &= ((x^{m/n})^l)^{1/l}\\
    &= (x^{ml/n})^{1/l}
      & \text{using $(x^a)^b=x^{ab}$}\\
    &= (x^{k})^{1/l}
      & \text{hypothesis: $k=ml/n$}\\
    &= x^{k/l}\\
    &= (x^{1/l})^k\\
\end{align*}
\end{solution}
